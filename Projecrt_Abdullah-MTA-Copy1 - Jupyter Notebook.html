      def get_data(week_nums):
          url = "http://web.mta.info/developers/data/nyct/turnstile/turnstile_{}.txt"
          dfs = []
          for week_num in week_nums:
              file_url = url.format(week_num)
              dfs.append(pd.read_csv(file_url))
          return pd.concat(dfs)
